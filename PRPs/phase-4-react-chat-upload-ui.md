### Step 3: Custom SSE hook — `client/src/hooks/useDocumentChat.ts`

This is the **frontend nerve center** (~50-70 lines). It:
1. Sends a POST to `/api/questions` with `{ question, documentId }`
2. Reads the SSE stream using `fetch()` + `ReadableStream`
3. Parses `event: <type>\ndata: <json>\n\n` format
4. Accumulates `text-delta` events into message content
5. Collects `citation` events into message citations array
6. Finalizes on `done` event

```typescript
import { useState, useCallback, useRef } from "react";
import type { ChatMessage, Citation } from "@/lib/types";

// Generate unique IDs for messages
let nextId = 0;
const generateId = () => `msg-${++nextId}`;

export function useDocumentChat() {
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [isLoading, setIsLoading] = useState(false);
  const abortRef = useRef<AbortController | null>(null);

  const sendMessage = useCallback(async (question: string, documentId: string) => {
    // Add user message
    const userMsg: ChatMessage = {
      id: generateId(),
      role: "user",
      content: question,
      citations: [],
      isStreaming: false,
    };
    const assistantId = generateId();
    const assistantMsg: ChatMessage = {
      id: assistantId,
      role: "assistant",
      content: "",
      citations: [],
      isStreaming: true,
    };
    setMessages(prev => [...prev, userMsg, assistantMsg]);
    setIsLoading(true);

    // Abort any previous stream
    abortRef.current?.abort();
    const controller = new AbortController();
    abortRef.current = controller;

    try {
      const res = await fetch("/api/questions", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ question, documentId }),
        signal: controller.signal,
      });

      if (!res.ok || !res.body) {
        throw new Error("Failed to start stream");
      }

      const reader = res.body.getReader();
      const decoder = new TextDecoder();
      let buffer = "";

      // Mutable accumulators — pushed into React state via setMessages
      let text = "";
      let citations: Citation[] = [];

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buffer += decoder.decode(value, { stream: true });
        const parts = buffer.split("\n\n");
        buffer = parts.pop() ?? "";

        for (const part of parts) {
          const lines = part.split("\n");
          let eventType = "";
          let data = "";
          for (const line of lines) {
            if (line.startsWith("event: ")) eventType = line.slice(7);
            else if (line.startsWith("data: ")) data = line.slice(6);
          }

          if (eventType === "text-delta" && data) {
            const parsed = JSON.parse(data);
            text += parsed.text;
          } else if (eventType === "citation" && data) {
            const parsed = JSON.parse(data) as Citation;
            citations = [...citations, parsed];
          } else if (eventType === "done") {
            // Stream complete
          }

          // Update assistant message in place
          setMessages(prev =>
            prev.map(m =>
              m.id === assistantId
                ? { ...m, content: text, citations, isStreaming: eventType !== "done" }
                : m
            )
          );
        }
      }
    } catch (err) {
      if ((err as Error).name !== "AbortError") {
        setMessages(prev =>
          prev.map(m =>
            m.id === assistantId
              ? { ...m, content: m.content || "An error occurred. Please try again.", isStreaming: false }
              : m
          )
        );
      }
    } finally {
      setIsLoading(false);
    }
  }, []);

  const clearMessages = useCallback(() => {
    abortRef.current?.abort();
    setMessages([]);
  }, []);

  return { messages, isLoading, sendMessage, clearMessages };
}
```

**Key design decisions:**
- Uses `fetch()` + `ReadableStream` (not `EventSource`) because `EventSource` only supports GET, and we need POST with a JSON body
- Accumulates text in a mutable `let` variable outside React state for performance — pushes into state after each SSE event
- Each assistant message has its own `id` for targeted state updates
- `abortRef` allows canceling an in-flight stream when a new question is sent
- `clearMessages` resets state when switching documents
- Error handling: on fetch failure or stream error, show error text in the assistant bubble

```typescript
import { useState, useEffect, useCallback } from "react";
import { AppLayout } from "@/components/layout/AppLayout";
import { ChatPanel } from "@/components/chat/ChatPanel";
import { UploadDropzone } from "@/components/upload/UploadDropzone";
import { DocumentList } from "@/components/upload/DocumentList";
import { useDocumentChat } from "@/hooks/useDocumentChat";
import { uploadDocument, getDocuments } from "@/lib/api";
import type { DocumentDto, Citation } from "@/lib/types";

export function App() {
  const [documents, setDocuments] = useState<DocumentDto[]>([]);
  const [selectedDoc, setSelectedDoc] = useState<DocumentDto | null>(null);
  const [activeCitation, setActiveCitation] = useState<Citation | null>(null);
  const { messages, isLoading, sendMessage, clearMessages } = useDocumentChat();

  // Load documents on mount
  useEffect(() => {
    getDocuments().then(setDocuments).catch(console.error);
  }, []);

  const handleUpload = useCallback(async (file: File) => {
    const response = await uploadDocument(file);
    // Refresh document list
    const docs = await getDocuments();
    setDocuments(docs);
    // Auto-select the newly uploaded document
    const newDoc = docs.find(d => d.id === response.id);
    if (newDoc) setSelectedDoc(newDoc);
  }, []);

  const handleSelectDoc = useCallback((doc: DocumentDto) => {
    if (doc.id !== selectedDoc?.id) {
      setSelectedDoc(doc);
      setActiveCitation(null);
      clearMessages();
    }
  }, [selectedDoc, clearMessages]);

  const handleSendMessage = useCallback((question: string) => {
    if (!selectedDoc) return;
    sendMessage(question, selectedDoc.id);
  }, [selectedDoc, sendMessage]);

  const handleCitationClick = useCallback((citation: Citation) => {
    setActiveCitation(citation);
  }, []);

  // Left panel: upload + document list + chat
  const leftPanel = (
    <div className="flex flex-col h-full">
      <div className="p-4 border-b space-y-3">
        <h1 className="text-lg font-semibold">AskMyPdf</h1>
        <UploadDropzone onUpload={handleUpload} />
        <DocumentList
          documents={documents}
          selectedId={selectedDoc?.id ?? null}
          onSelect={handleSelectDoc}
        />
      </div>
      <div className="flex-1 overflow-hidden">
        <ChatPanel
          messages={messages}
          isLoading={isLoading}
          hasDocument={selectedDoc !== null}
          onSendMessage={handleSendMessage}
          onCitationClick={handleCitationClick}
        />
      </div>
    </div>
  );

  // Right panel: PDF viewer placeholder (Phase 5)
  const rightPanel = (
    <div className="flex h-full items-center justify-center bg-muted/30 text-muted-foreground">
      {selectedDoc ? (
        <p>PDF Viewer — {selectedDoc.fileName} (Phase 5)</p>
      ) : (
        <p>Upload and select a document to view it here</p>
      )}
    </div>
  );

  return <AppLayout leftPanel={leftPanel} rightPanel={rightPanel} />;
}
```

**State management:**
- `documents` — list of all uploaded docs (refreshed after each upload)
- `selectedDoc` — currently active document (determines which doc to send to `/api/questions`)
- `activeCitation` — set when a citation chip is clicked (Phase 5 wires this to PDF viewer)
- `messages`, `isLoading` — from `useDocumentChat` hook

## SSE Event Format (from backend)

The backend streams events in this exact format:

```
event: text-delta
data: {"text":"chunk of answer text"}

event: citation
data: {"documentId":"abc123","documentName":"doc.pdf","pageNumber":3,"citedText":"Founded in 2015","highlightAreas":[{"pageIndex":2,"left":10.5,"top":20.3,"width":80.1,"height":1.2}]}

event: done
data: {}
```

- Events separated by `\n\n`
- Each event has `event: <type>\n` and `data: <json>\n` lines
- `text-delta` events arrive frequently (every few tokens) — accumulate into full text
- `citation` events arrive inline as Claude cites sources — collect into array
- `done` event signals stream complete

## Risks

| Risk | Mitigation |
|------|------------|
| **shadcn CLI fails or generates incompatible code** | The CLI is well-tested with React 18 + Tailwind 3. If it fails, manually copy component code from shadcn/ui docs. The components are just files — no runtime dependency. |
| **SSE parsing edge cases** — partial events, split across chunks | Buffer incoming data, split on `\n\n`, keep remainder for next chunk. This is the standard approach and handles partial reads correctly. |
